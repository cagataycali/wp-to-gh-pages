    Yukarıda index.php dosyasının içindekiler yazmakta. 
    
    
    <?php
    function dort_islem($sayi1,$sayi2,$islem){
    
        switch($islem){
            case "topla";
                $sonuc=$sayi1+$sayi2;
                break;
            case "cikar";
                $sonuc=$sayi1-$sayi2;
                break;
            case "bol";
                $sonuc=$sayi1/$sayi2;
                break;
            case "carp";
                $sonuc=$sayi1*$sayi2;
                break;
        }
    
    
    
        echo $sonuc;
    
    
    
    
    }
    
    
    
    ?>

Yukarıda matfunc.php içersindekiler yer almakta.  
    
    
    <?php
    function html_ust($t){
        echo'
                <html>
                <head>
                <meta http-equiv="content-type" content="text/html" charset="UTF-8" />
                <title>'.$t.'</title>
                </head>
                <body>
        ';
    }
    
    
    function html_alt(){
        echo'</body></html>';
    }
