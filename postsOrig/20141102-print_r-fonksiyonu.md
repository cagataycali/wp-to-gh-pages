# print_r() Fonksiyonu

print_r() fonksiyonu, bir dizinin içeriğini  ekrana yazdırır.   Örnek ; 
    
    
    <?php
    $dizi=array("Çağatay", "Çalı", "2014");
    print_r($dizi);
    ?>
